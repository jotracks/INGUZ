/* =========================
   PUSH GITHUB (FIX)
========================= */
async function pushToGitHub(data) {
  if (!GH_TOKEN || syncPending) return;

  syncPending = true;

  try {
    if (!ghSha) {
      const res = await fetch(GH_API, {
        headers: { Authorization: `token ${GH_TOKEN}` }
      });

      if (res.ok) {
        const json = await res.json();
        ghSha = json.sha;
      }
    }

    const body = {
      message: 'update',
      content: btoa(unescape(encodeURIComponent(JSON.stringify(data, null, 2))))
    };

    // 🔥 FIX: solo mandar sha si existe
    if (ghSha) body.sha = ghSha;

    const res = await fetch(GH_API, {
      method: 'PUT',
      headers: {
        Authorization: `token ${GH_TOKEN}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(body)
    });

    if (!res.ok) {
      console.error(await res.text());
    } else {
      const json = await res.json();
      ghSha = json.content.sha;
    }

  } catch (e) {
    console.error(e);
  }

  syncPending = false;
}

/* =========================
   PULL
========================= */
async function pullFromGitHub() {
  if (!GH_TOKEN) return;

  try {
    const res = await fetch(GH_API);

    if (!res.ok) return;

    const json = await res.json();
    ghSha = json.sha;

    const decoded = decodeURIComponent(escape(atob(json.content)));
    const remote = JSON.parse(decoded);

    localStorage.setItem('inguz_data', JSON.stringify(remote));
    render();

  } catch (e) {
    console.error(e);
  }
}

/* =========================
   TOKEN
========================= */
function guardarToken() {
  const token = document.getElementById('setup-token-inp').value.trim();
  const err = document.getElementById('setup-err');

  err.style.display = 'none';

  if (!token) {
    err.textContent = 'Pegá un token válido';
    err.style.display = 'block';
    return;
  }

  localStorage.setItem(GH_TOKEN_KEY, token);
  GH_TOKEN = token;

  activarHome();
  pullFromGitHub();
}

/* =========================
   UI
========================= */
function activarSetup() {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-setup').classList.add('active');
}

function activarHome() {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-home').classList.add('active');
}

function agregarEjemplo() {
  const d = getData();

  if (!d.historial) d.historial = [];

  d.historial.push({
    fecha: new Date().toISOString(),
    ejercicio: "Press banca",
    peso: Math.floor(Math.random()*100)
  });

  saveData(d);
}

function render() {
  document.getElementById('debug').textContent =
    JSON.stringify(getData(), null, 2);
}

function resetApp() {
  localStorage.clear();
  location.reload();
}

document.addEventListener('DOMContentLoaded', () => {
  const token = localStorage.getItem(GH_TOKEN_KEY);

  if (!token) {
    activarSetup();
  } else {
    GH_TOKEN = token;
    activarHome();
    pullFromGitHub();
  }
